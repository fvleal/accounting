```typescript
export function uploadPhoto(file: File): Promise<Account> {
  const formData = new FormData();
  formData.append('file', file);
  return apiClient
    .post<ApiResponse<Account>>('/accounts/me/photo', formData, {
      headers: { 'Content-Type': 'multipart/form-data' },
    })
    .then((r) => r.data.data);
}
```

```typescript
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { updateAccount } from '../api/accounts';
import type { Account } from '../types/account';

export function useUpdateAccount() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: (data: { name?: string; birthDate?: string }) => updateAccount(data),
    onSuccess: (account: Account) => {
      queryClient.setQueryData(['account', 'me'], account);
    },
  });
}
```
